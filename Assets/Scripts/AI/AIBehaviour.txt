tree("Root")
	parallel
		repeat mute tree("Patrol")

tree("Patrol")
	fallback
		tree("Flee")
		while not IsHealthLessThan(50.0)
			fallback
				tree("Attack")
				tree("Wander")

tree("Flee")
	while IsHealthLessThan(50.0)
		fallback
			while not HasTargetInRange
				sequence
					tree("Wander")
			while HasTargetInRange
				fallback
					while InDanger(15.0)
						sequence
							TakeCover
							MoveDestination
		
tree("Wander")
	while not HasTargetInRange
		sequence
			StopFire
			PickRandomDestination
			MoveDestination

tree("Attack")
	while HasTargetInRange
		sequence
			while not DistanceToShoot(15.0)
				GetDistance
