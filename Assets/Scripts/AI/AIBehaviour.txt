tree("Root")
	parallel
		repeat mute tree("Patrol")

tree("Patrol")
	fallback
		tree("Flee")
		while not IsHealthLessThan(50.0)
			fallback
				tree("Attack")
				tree("Wander")

tree("Flee")
	while IsHealthLessThan(50.0)
		fallback
			while not SeeEnemy
				sequence
					tree("Wander")
			while SeeEnemy
				fallback
					while InDanger(15.0)
						sequence
							TakeCover
							MoveDestination
		
tree("Wander")
	while not SeeEnemy
		sequence
			StopFire
			PickRandomDestination
			MoveDestination

tree("Attack")
	while SeeEnemy
		parallel
			repeat
				sequence
					AlignTank
					Fire
			repeat
				sequence
					GetDistance
					Wait(0.1)
